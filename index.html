            // 導航欄捲動效果 (Navbar scroll effect)

            const navbar = document.getElementById('navbar');

            const handleScroll = () => {

                if (window.scrollY > 50) {

                    navbar.classList.remove('bg-transparent', 'py-6');

                    navbar.classList.add('bg-neutral-950/90', 'backdrop-blur-md', 'border-b', 'border-white/10', 'py-4');

                } else {

                    navbar.classList.remove('bg-neutral-950/90', 'backdrop-blur-md', 'border-b', 'border-white/10', 'py-4');

                    navbar.classList.add('bg-transparent', 'py-6');

                }

            };

            window.addEventListener('scroll', handleScroll);

            

            // 手機菜單邏輯 (Mobile Menu Logic)

            const menuButton = document.getElementById('mobile-menu-button');

            const mobileMenu = document.getElementById('mobile-menu');

            const menuIcon = document.getElementById('menu-icon');

            const xIcon = document.getElementById('x-icon');


            const toggleMobileMenu = () => {

                const isOpen = mobileMenu.classList.contains('flex');

                if (isOpen) {

                    mobileMenu.classList.remove('flex');

                    mobileMenu.classList.add('hidden');

                    menuIcon.classList.remove('hidden');

                    xIcon.classList.add('hidden');

                } else {

                    mobileMenu.classList.remove('hidden');

                    mobileMenu.classList.add('flex');

                    menuIcon.classList.add('hidden');

                    xIcon.classList.remove('hidden');

                }

            };


            menuButton.addEventListener('click', toggleMobileMenu);


            // 點擊菜單連結後關閉手機菜單 (Close mobile menu after clicking a link)

            mobileMenu.querySelectorAll('a').forEach(link => {

                link.addEventListener('click', toggleMobileMenu);

            });

        });
